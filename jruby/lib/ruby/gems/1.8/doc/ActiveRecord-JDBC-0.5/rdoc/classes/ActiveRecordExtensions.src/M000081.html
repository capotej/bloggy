    def mysql_connection(config)
      if config[:socket]
        warn "AR-JDBC MySQL on JRuby does not support sockets"
      end
      config[:port] ||= 3306
      config[:url] ||= "jdbc:mysql://#{config[:host]}:#{config[:port]}/#{config[:database]}"
      config[:driver] = "com.mysql.jdbc.Driver"
      jdbc_connection(config)
    end
