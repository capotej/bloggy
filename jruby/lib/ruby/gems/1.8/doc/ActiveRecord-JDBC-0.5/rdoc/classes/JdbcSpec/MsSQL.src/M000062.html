          def change_order_direction(order)
            order.split(",").collect {|fragment|
              case fragment
                when  /\bDESC\b/i     then fragment.gsub(/\bDESC\b/i, "ASC")
                when  /\bASC\b/i      then fragment.gsub(/\bASC\b/i, "DESC")
                else                  String.new(fragment).split(',').join(' DESC,') + ' DESC'
              end
            }.join(",")
          end
