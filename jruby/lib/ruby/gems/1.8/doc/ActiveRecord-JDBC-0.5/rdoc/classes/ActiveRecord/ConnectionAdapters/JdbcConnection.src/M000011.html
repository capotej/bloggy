      def initialize(config)
        @config = config.symbolize_keys!
        if @config[:jndi]
          configure_jndi
        else
          configure_jdbc
        end
        set_native_database_types
        @stmts = {}
      rescue Exception => e
        raise "The driver encountered an error: #{e}"
      end
