      def exists?(id_or_conditions)
        connection.select_all(
          construct_finder_sql(
            :select     => "#{quoted_table_name}.#{primary_key}",
            :conditions => expand_id_conditions(id_or_conditions),
            :limit      => 1
          ),
          "#{name} Exists"
        ).size > 0
      end
