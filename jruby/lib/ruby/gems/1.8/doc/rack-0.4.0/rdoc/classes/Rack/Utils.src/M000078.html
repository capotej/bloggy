    def parse_query(qs, d = '&;')
      params = {}
      
      (qs || '').split(/[#{d}] */n).each do |p|
        k, v = unescape(p).split('=', 2)
        
        if cur = params[k]
          if cur.class == Array
            params[k] << v
          else
            params[k] = [cur, v]
          end
        else
          params[k] = v
        end
      end
      
      return params
    end
