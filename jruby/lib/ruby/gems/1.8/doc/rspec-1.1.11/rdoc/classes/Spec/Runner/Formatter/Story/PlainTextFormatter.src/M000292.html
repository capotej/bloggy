          def scenario_started(story_title, scenario_name)
            @current_scenario_name = scenario_name
            @scenario_already_failed = false
            @scenario_head_text = "\n\n  Scenario: #{scenario_name}"
            @scenario_body_text = ""
            @scenario_ok = true
            @scenario_pending = false
            @scenario_failed = false
          end
