      def order!(argv, &blk)
        @argv = argv.dup
        @argv = (@argv.empty? && Spec.spec_command?) ? ['--help'] : @argv 
        @options.argv = @argv.dup
        return if parse_generate_options
        return if parse_drb
        
        super(@argv) do |file|
          @options.files << file
          blk.call(file) if blk
        end

        @options
      end
