        def dump_pending
          unless @pending_examples.empty?
            @output.puts
            @output.puts "Pending:"
            @pending_examples.each do |pending_example|
              @output.puts "#{pending_example[0]} (#{pending_example[1]})" 
              @output.puts "  Called from #{pending_example[2]}"
            end
          end
          @output.flush
        end
