          def step_failed(type, description, *args)
            found_step(type, description, true, @scenario_pending, *args)
            if @scenario_pending
              @scenario_body_text +=  yellow(" (SKIPPED)")
            else
              @scenario_body_text +=  red(@scenario_ok ? " (FAILED)" : " (SKIPPED)")
            end
            @scenario_ok = false
          end
