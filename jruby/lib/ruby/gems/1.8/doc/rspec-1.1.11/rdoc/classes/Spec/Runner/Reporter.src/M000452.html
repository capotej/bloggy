      def example_finished(example, error=nil)
        @examples << example
        
        if error.nil?
          example_passed(example)
        elsif Spec::Example::ExamplePendingError === error
          example_pending(example, error.pending_caller, error.message)
        else
          example_failed(example, error)
        end
      end
