          def collected_steps(steps)
            unless steps.empty?
              @output.puts "      <ul id=\"stock_steps\" style=\"display: none;\">"
              steps.each do |step|
                @output.puts "        <li>#{step}</li>"
              end
              @output.puts "      </ul>"
            end
          end
