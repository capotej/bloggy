literal(v)

Handle microseconds for Time and DateTime values, as well as PostgreSQL specific boolean values and string escaping.

[Source]

     # File lib/sequel_core/adapters/shared/postgres.rb, line 477
477:       def literal(v)
478:         case v
479:         when LiteralString
480:           v
481:         when SQL::Blob
482:           db.synchronize{|c| "E'#{c.escape_bytea(v)}'"}
483:         when String
484:           db.synchronize{|c| "'#{c.escape_string(v)}'"}
485:         when Time
486:           "#{v.strftime(PG_TIMESTAMP_FORMAT)}.#{sprintf("%06d",v.usec)}'"
487:         when DateTime
488:           "#{v.strftime(PG_TIMESTAMP_FORMAT)}.#{sprintf("%06d", (v.sec_fraction * 86400000000).to_i)}'"
489:         when TrueClass
490:           BOOL_TRUE
491:         when FalseClass
492:           BOOL_FALSE
493:         else
494:           super
495:         end
496:       end
