    def add_limit_offset!(sql, options)
      if options[:limit]
        limit = "FIRST #{options[:limit]}"
        # SKIP available only in IDS >= 10
        offset = (@@db_major_version >= 10 && options[:offset]?
                  "SKIP #{options[:offset]}" : "")
        sql.sub!(/^select /i, "SELECT #{offset} #{limit} ")
      end
      sql
    end
