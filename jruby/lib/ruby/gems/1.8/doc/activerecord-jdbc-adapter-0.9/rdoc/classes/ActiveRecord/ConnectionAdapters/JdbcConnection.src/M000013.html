      def adapter=(adapt)
        @adapter = adapt
        @tps = {}
        @native_types.each_pair {|k,v| @tps[k] = v.inject({}) {|memo,kv| memo.merge({kv.first => (kv.last.dup rescue kv.last)})}}
        adapt.modify_types(@tps)
      end
