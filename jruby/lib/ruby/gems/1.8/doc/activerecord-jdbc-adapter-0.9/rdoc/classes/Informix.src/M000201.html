    def quote(value, column = nil)
      if column && [:binary, :text].include?(column.type)
        # LOBs are updated separately by an after_save trigger.
        "NULL"
      elsif column && column.type == :date
        "'#{value.mon}/#{value.day}/#{value.year}'"
      else
        super
      end
    end
