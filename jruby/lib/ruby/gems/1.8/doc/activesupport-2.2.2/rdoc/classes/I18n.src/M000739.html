    def translate(key, options = {})
      locale = options.delete(:locale) || I18n.locale
      backend.translate(locale, key, options)
    rescue I18n::ArgumentError => e
      raise e if options[:raise]
      send(@@exception_handler, e, locale, key, options)
    end
