  def cache_decr(server, cache_key, amount)
    socket = server.socket
    socket.write "decr #{cache_key} #{amount}\r\n"
    text = socket.gets
    raise_on_error_response! text
    return nil if text == "NOT_FOUND\r\n"
    return text.to_i
  end
