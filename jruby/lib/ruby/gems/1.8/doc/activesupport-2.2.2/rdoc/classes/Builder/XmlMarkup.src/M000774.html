    def instruct!(directive_tag=:xml, attrs={})
      _ensure_no_block block_given?
      if directive_tag == :xml
        a = { :version=>"1.0", :encoding=>"UTF-8" }
        attrs = a.merge attrs
      end
      _special(
      "<?#{directive_tag}",
      "?>",
      nil,
      attrs,
      [:version, :encoding, :standalone])
    end
