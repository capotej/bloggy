    def group!(*args, &block)
      args.each do |arg|
        if arg.is_a?(Symbol)
          instance_eval(&@library[arg])
        else
          instance_eval(&arg)
        end
        _text ', ' unless arg == args.last
      end
      if block
        _css_block(block)
        _unify_block
      end
    end
