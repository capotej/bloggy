      def unmemoize_all
        methods.each do |m|
          if m.to_s =~ /^_unmemoized_(.*)/
            ivar = MEMOIZED_IVAR.call($1)
            instance_variable_get(ivar).clear if instance_variable_defined?(ivar)
          end
        end
      end
