    def declare!(inst, *args, &block)
      _indent
      @target << "<!#{inst}"
      args.each do |arg|
        case arg
        when String
          @target << %{ "#{arg}"} # " WART
        when Symbol
          @target << " #{arg}"
        end
      end
      if block_given?
        @target << " ["
        _newline
        _nested_structures(block)
        @target << "]"
      end
      @target << ">"
      _newline
    end
